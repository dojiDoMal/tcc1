\newglossaryentry{view-frustum}{
	name={Viewing Frustum},
	description={região do espaço no mundo modelado que pode aparecer na tela; campo de visão de um sistema de câmera virtual em perspectiva}
}
\newglossaryentry{multithread}{
	name={Multithread},
	description={capacidade que o sistema operacional possui de executar vários conjuntos de tarefas simultaneamente sem que uma interfira na outra}
}
\newglossaryentry{scripts}{
	name={Scripts},
	description={programas escritos para um sistema de tempo de execução especial que automatiza a execução de tarefas}
}
\newglossaryentry{realidade-virtual}{
	name={Realidade Virtual},
	description={ambiente gerado por computador com cenas e objetos que parecem reais, fazendo com que os usuários se sintam imersos nessa realidade}
}
\newglossaryentry{realidade-aumentada}{
	name={Realidade Aumentada},
	description={integração de elementos virtuais a visualizações do mundo real normalmente através de uma câmera}
}
